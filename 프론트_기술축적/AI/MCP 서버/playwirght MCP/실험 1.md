    const fuelTypeCheckbox = page.getByText('연료타입').locator('..').locator('input[type="checkbox"]');
    await freezerTempCheckbox.check({ force: true });
    await fuelTypeCheckbox.check({ force: true });

    // 4. '>' 아이콘 버튼 클릭 (항목 추가)
    await page.locator('button:has(svg.tableSetting_icon__1h9hn7p0)').first().click();

    // 5. 항목이 오른쪽(추가할 수 있는 항목)으로 이동했는지 확인
    // (예: "추가할 수 있는 항목 (2)" 텍스트가 보이는지)
    await expect(page.getByText('추가할 수 있는 항목 (2)')).toBeVisible();

    // 6. 팝업 닫기 (취소/적용 등)
    await page.getByRole('button', { name: '취소' }).click();
    await expect(popup).toBeHidden();
  });
});

import { test, expect } from '@playwright/test';

const BASE_URL = 'https://172.28.5.93:8443/web/client/vehicles/list';

test.use({ storageState: 'playwright/.auth/user.json' });

    const fuelTypeCheckbox = page

    await expect(page.getByText('추가할 수 있는 항목 (2)')).toBeVisible();

    await page.getByRole('button', { name: '취소' }).click();
    await expect(popup).toBeHidden();

  test('테이블 항목 설정 팝업에서 드래그&드롭(항목 순서 변경) 기능', async ({ page }) => {
    await page
      .locator('div')
      .filter({ hasText: /^차량 타입차량 이용 상태이상 상태$/ })
      .locator('button')
      .nth(3)
      .click();
    const popup = page.locator(
      'div[role="dialog"], div:has-text("테이블 항목 설정")',
    );
    await expect(popup).toBeVisible();

    // 예시: "총 주행 거리"를 "냉동기 온도" 위로 드래그
    const from = popup.getByText('총 주행 거리');
    const to = popup.getByText('냉동기 온도');
    await from.dragTo(to);

    // 순서가 바뀌었는지 확인 (간단히 텍스트 순서로 검증)
    const items = await popup.locator('div').filter({ hasText: '총 주행 거리' }).allTextContents();
    expect(items.some((t) => t.includes('총 주행 거리'))).toBeTruthy();
  });

  test('테이블 항목 설정 팝업 저장/초기화 기능', async ({ page }) => {
    await page
      .locator('div')
      .filter({ hasText: /^차량 타입차량 이용 상태이상 상태$/ })

      .locator('div')
      .filter({ hasText: /^차량 타입차량 이용 상태이상 상태$/ })
      .locator('button')
      .nth(3)
      .click();
    await expect(popup).toBeVisible();

    // 초기화 후 팝업이 닫히거나, 상태가 원래대로 돌아가는지 확인
    await expect(popup).toBeHidden();
  });

아래와 같이 **드래그&드롭, 저장, 초기화** 기능까지 모두 Playwright 테스트 코드로 추가했습니다.

- **드래그&드롭:**
